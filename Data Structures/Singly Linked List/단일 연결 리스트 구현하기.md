연결 리스트 (Linked List) 구현하기
======

<h2>연결 리스트의 구조</h2>
<hr/>

* 연결 리스트는 노드(Node)를 최소 단위로 하는 자료구조이다.
* Node는 data와 다음 노드를 가리키는 포인터 정보를 가지고 있다.
* 우선 연결 리스트를 간단히 구현해보자.
<hr/>

<h2>#1. 기본적인 연결 리스트의 구현</h2>

```C
#include <stdio.h>
#include <stdlib.h>

// 노드의 설계
typedef struct Node {
    int data;
    struct Node *next;
}Node;

//첫 노드를 가리키는 head 전역 변수
Node *head;

// 데이터를 마지막에 추가하는 listAdd 함수
int listAdd(int data){
    Node *node = calloc(1, sizeof(Node));

    //동적 할당 실패 여부 검사
    if(node == NULL){
        fprintf(stderr, "listAdd: calloc failed.\n");
        return -1;
    }

    node->data = data;

    // 연결 리스트가 비어있는 경우
    if(head == NULL){
        head = node;
        return 0;
    }

    // 연결 리스트에 노드가 존재하는 경우
    // head와 같은 값을 저장하는 cur 포인터변수 생성
    Node *cur = head;

    // cur 포인터가 리스트의 마지막 노드가 될 때 까지 loop를 돌린다.
    while(cur->next != NULL){
        cur = cur->next;
    }

    // 마지막 노드가 된 cur가 새로운 node를 가리키도록 한다.
    cur->next = node;
    return 0;
}

// debugging code
void listDisplay(){
    system("cls");
    printf("[head]");
    for(Node *node = head; node != NULL; node = node->next) {
        printf("->[%2d]", node->data);
    }
    getchar();
}

int main(){
   listDisplay();
	for (int i = 0; i < 5; i++) {
		listAdd(i + 1);
		listDisplay();
	}
	return 0; 
}
```

* 위 자료구조는 listAdd 함수의 데이터 삽입 알고리즘이 이원화 되어있다는 단점이 있다.
* 이를 해결하기 위해 더미 헤드(Dummy Head)를 삽입하자.
<hr/>

<h2>#2. listAdd 함수 알고리즘의 일반화 및 더미 헤드 삽입</h2>

```C
#include <stdlib.h>
#include <stdio.h>

typedef struct Node {
	int data;
	struct Node* next;
} Node;
Node* head;

int listAdd(int data) {
	Node* node = calloc(1, sizeof(Node));
	if (node == NULL) {
		perror("listAdd");
		return -1;
	}
	node->data = data;

    // head가 NULL인지 검사할 필요가 없다.

    Node* cur = head;
	while (cur->next != NULL)
		cur = cur->next;
	cur->next = node;
	return 0;
}

// listDisplay 함수는 기존과 동일

int main(){

    // 더미 헤드 추가
    // 더미 헤드는 메모리를 생성해주고, 0또는 NULL로 초기화해주는
    // calloc 함수를 이용하여 할당한다.
    head = calloc(1, sizeof(Node));
    listDisplay();
	for (int i = 0; i < 5; i++) {
		listAdd(i + 1);
		listDisplay();
	}
	return 0;
}
```
* 위 자료구조는 사용자가 초기화를 직접 해야한다는 단점이 있다.
* 이를 해결하기 위해 초기화 함수를 제공하자.

<h2>#3. 자료구조 초기화 함수 listInitialize의 구현</h2>

* 위 코드는 각 함수에서 마지막 노드인지를 판별하기 위해 NULL과 비교하고 있다. 
* 이는 잠재적으로 NULL 참조가 발생할 위험이 있으므로, tail로 해결한다.
* tail은 항상 리스트의 마지막 노드가 되며, 이전의 더미 헤드와 마찬가지로 더미 테일(Dummy tail)을 사용하도록 하자.

<h2>#4. 더미 테일의 도입</h2>

```C
#include <stdio.h>
#include <stdlib.h>

typedef struct Node{
    int data;
    struct Node *next;
}Node;

Node *head;
Node *tail;

int listInitialize() {
	head = malloc(sizeof(Node));
	if (head == NULL) {
		perror("listInitialize");
		return -1;
	}

	tail = malloc(sizeof(Node));
	if (tail == NULL) {
		perror("listInitialize");
		free(head);
		return -1;
	}
	head->next = tail;
	tail->next = tail;
	return 0;
}

int listAdd(int data) {
	Node* node = malloc(sizeof(Node));
	if (node == NULL) {
		perror("listAdd");
		return -1;
	}
	node->data = data;
	node->next = tail;

	Node* cur = head;
    
    // while문이 cur->next가 NULL이 아닌
    // tail을 가리킬 때 까지 돌게 한다.
	while (cur->next != tail)
		cur = cur->next;
	cur->next = node;
	return 0;
}

void listDisplay() {
	system("cls");
	printf("[head]");
	for (Node* node = head->next; node != tail; node = node->next)
		printf("->[%2d]", node->data);
	printf("->[tail]");
	getchar();
}

int main() {
	listInitialize();

	listDisplay();
	for (int i = 0; i < 5; i++) {
		listAdd(i + 1);
		listDisplay();
	}
	return 0;
}
```

* 이제 다음으로 해야할 작업은 자료구조에 내부적으로 사용되는   
  Node들, 더미 head, 더미 tail에 대해 메모리 해제를 하여   
  메모리 누수를 막는 작업이다.
* 메모리 해제는 listFinalize 함수를 통해 구현해보자.

<h2>#5. 동적 메모리 해제 함수 구현</h2>

```C
// 기타 코드 모두 동일

int listFinalize(){
    if(head == NULL || tail == NULL){
        fprintf(stderr, "listFinalize:head or tail is NULL.");
        return -1;
    }

    while(head->next != tail){
        // target에 head가 가리키는 노드의 주소 저장
        Node *target = head->next;
        // head의 next에 target이 가리키는 노드의 주소 저장
        // 즉, head->next가 두 개 앞선 노드를 가리키게 한다.
        head->next = target->next;
        free(target);
    }
    free(head);
    free(tail);
    return 0;
}

int main(){
    listInitialize();

    listDisplay();
    for(int i = 0; i < 5; i++){
        listAdd(i + 1);
        listDisplay();
    }
    listFinalize();
    return 0;
}
```

<h2>#7. 특정 위치에 데이터를 삽입하는 listInsert함수의 구현</h2>

```C
// 기타 코드 동일
int listCount(){
    return count;
}

int listCount() {
	return count;
}

int listInsert(int index, int data) {

	if (listCount() == 0) {
		fprintf(stderr, "listSet:list is empty.\n");
		return -1;
	}

	if (index < 0 || index >= listCount()) {
		fprintf(stderr, "listSet:index is out of bound.\n");
		return -1;
	}

	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		fprintf(stderr, "listSet:malloc failed.\n");
		return -1;
	}
	node->data = data;

	// prev 포인터에 head를 대입한다.
	Node *prev = head;

	//prev가 (index-1)번 째 노드까지 가도록 for루프를 돌린다.
	// (index-1)번 째인 이유는 prev가 초기에 head->next가 아닌 head로 초기화되었기 때문이다.
	for (int i = 0; i < index; i++) {
		prev = prev->next;
	}

	// 추가할 노드인 node의 next가 가리키는 곳이 prev의 next와 동일하게 한다.
	node->next = prev->next;
	// prev의 next가 node를 가리키게 한다.
	prev->next = node;
	// 노드가 추가되었으므로 count값을 증가시킨다.
	++count;
	return 0;
}

int listAdd(int data) {
	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		perror("listAdd");
		return -1;
	}
	node->data = data;
	node->next = tail;

	Node *cur = head;
	while (cur->next != tail)
		cur = cur->next;
	cur->next = node;
	count++; 
	return 0;
}
```

* 이제는 특정 index의 데이터를 읽어오는 함수를 구현해보자.

<h2>#9. 특정 위치의 노드를 제거하는 listRemove함수의 구현</h2>

```C
int listRemove(int index, int *outData) {
	if (listCount() == 0) {
		fprintf(stderr, "listGet:list is empty.\n");
		return -1;
	}

	if (index < 0 || index >= listCount()) {
		fprintf(stderr, "listGet:index is out of bound.\n");
		return -1;
	}

	if (outData == NULL) {
		fprintf(stderr, "listGet:outData is NULL.\n");
		return -1;
	}

	Node *prev = head;

	for (int i = 0; i < index; i++) {
		prev = prev->next;
	}
	Node *delNode = prev->next;
	*outData = delNode->data;
	prev->next = delNode->next;
	return 0;
}
```

* 위 자료구조는 이제 필수적인 기능을 모두 갖추었다.
* 하지만 전역 심볼을 사용하기에 2개 이상의 자료구조를 만들 수 없다는 단점이 있다.
* 또한, head, tail, count는 많은 함수들에서 사용되고, 여러개의 자료구조가 있다면 각각 다르므로   
  이들을 추상화하여 묶어주는 것이 좋다.

<h2>#10. 데이터의 추상화</h2>

```C
#include <stdlib.h>
#include <stdio.h>
#include <memory.h>
typedef struct Node {
	int data;
	struct Node *next;
} Node;

// head, tail, count는 List라는 타입으로 묶어 추상화한다.
typedef struct List {
	Node *head;
	Node *tail;
	int count;
}List;

int listCount(List *list) {
	return list->count;
}

int listRemove(List *list, int index, int *outData) {
	if (list->count == 0) {
		fprintf(stderr, "listGet:list is empty.\n");
		return -1;
	}

	if (index < 0 || index >= list->count) {
		fprintf(stderr, "listGet:index is out of bound.\n");
		return -1;
	}

	if (outData == NULL) {
		fprintf(stderr, "listGet:outData is NULL.\n");
		return -1;
	}

	Node *prev = list->head;

	for (int i = 0; i < index; i++) {
		prev = prev->next;
	}
	Node *delNode = prev->next;
	*outData = delNode->data;
	prev->next = delNode->next;
	return 0;
}

int listGet(List *list, int index, int *outData) {
	if (list->count == 0) {
		fprintf(stderr, "listGet:list is empty.\n");
		return -1;
	}

	if (index < 0 || index >= list->count) {
		fprintf(stderr, "listGet:index is out of bound.\n");
		return -1;
	}

	if (outData == NULL) {
		fprintf(stderr, "listGet:outData is NULL.\n");
		return -1;
	}

	Node *node = list->head->next;
	for (int i = 0; i < index; i++) {
		node = node->next;
	}

	*outData = node->data;
	return 0;

}
int listInsert(List *list, int index, int data) {

	if (list->count == 0) {
		fprintf(stderr, "listInsert:list is empty.\n");
		return -1;
	}

	if (index < 0 || index >= list->count) {
		fprintf(stderr, "listInsert:index is out of bound.\n");
		return -1;
	}

	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		fprintf(stderr, "listInsert:malloc failed.\n");
		return -1;
	}
	node->data = data;

	Node *prev = list->head;

	for (int i = 0; i < index; i++) {
		prev = prev->next;
	}

	node->next = prev->next;
	prev->next = node;
	++(list->count);
	return 0;
}

int listSet(List *list, int index, int newData, int *oldData) {
	if (list->count == 0) {
		fprintf(stderr, "listSet:list is empty.\n");
		return -1;
	}

	if (oldData == NULL) {
		fprintf(stderr, "listSet:oldData is NULL.\n");
		return -1;
	}

	if (index < 0 || index >= list->count) {
		fprintf(stderr, "listSet:index is out of bound.\n");
		return -1;
	}

	Node *cur = list->head->next;
	for (int i = 0; i < index; i++)
		cur = cur->next;
	*oldData = cur->data;
	cur->data = newData;
	return 0;
}

List *listInitialize() {

	List *list = calloc(1, sizeof(List));

	if (list == NULL) {
		fprintf(stderr, "listInitialize:calloc failed.\n");
		return NULL;
	}

	Node *head = malloc(sizeof(Node));
	if (head == NULL) {
		perror("listInitialize");
		return NULL;
	}

	Node *tail = malloc(sizeof(Node));
	if (tail == NULL) {
		perror("listInitialize");
		free(head);
		return NULL;
	}
	list->head = head;
	list->tail = tail;
	head->next = tail;
	tail->next = tail;
	list->count = 0;
	return list;
}

int listFinalize(List *list) {
	if (list == NULL) {
		fprintf(stderr, "listFinalize:list is NULL\n");
		return -1;
	}

	if (list->head == NULL || list->tail == NULL) {
		fprintf(stderr, "listFinalize: head or tail is NULL.\n");
		return -1;
	}

	while (list->head->next != list->tail) {
		Node *target = list->head->next;
		list->head->next = target->next;
		free(target);
	}
	free(list->head);
	free(list->tail);
	free(list);
	return 0;
}
int listAdd(List *list, int data) {
	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		perror("listAdd");
		return -1;
	}
	node->data = data;
	node->next = list->tail;

	Node *cur = list->head;
	while (cur->next != list->tail)
		cur = cur->next;
	cur->next = node;
	++(list->count);
	return 0;
}

void listDisplay(const List *list) {
	if (list == NULL) {
		fprintf(stderr, "listDisplay:list is NULL.\n");
		return;
	}

	system("cls");
	printf("[head]");
	for (Node *node = list->head->next; node != list->tail; node = node->next)
		printf("->[%2d]", node->data);
	printf("->[tail]");
	getchar();
}
```

* 위 자료구조는 Node가 int형 데이터만 담을 수 있기 때문에 마지막으로 Node가 모든 타입의 데이터를 담을 수 있도록 수정하자.

<h2>#11. 모든 타입을 담을 수 있도록 자료구조 재구현</h2>

	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		fprintf(stderr, "listInsert:malloc failed.\n");
		return -1;
	}
	node->data = data;

	Node *prev = list->head;

	for (int i = 0; i < index; i++) {
		prev = prev->next;
	}

	node->next = prev->next;
	prev->next = node;
	++(list->count);
	return 0;
}

void *listSet(List *list, int index, void *newData) {
	if (list == NULL) {
		fprintf(stderr, "listSet:list is NULL");
		return NULL;
	}
	
	if (list->count == 0) {
		fprintf(stderr, "listSet:list is empty.\n");
		return NULL;
	}

	if (index < 0 || index >= list->count) {
		fprintf(stderr, "listSet:index is out of bound.\n");
		return NULL;
	}

	Node *cur = list->head->next;
	for (int i = 0; i < index; i++)
		cur = cur->next;
	void * oldData = cur->data;
	cur->data = newData;
	return oldData;
}

List *listInitialize() {

	List *list = calloc(1, sizeof(List));

	if (list == NULL) {
		fprintf(stderr, "listInitialize:calloc failed.\n");
		return NULL;
	}

	Node *head = malloc(sizeof(Node));
	if (head == NULL) {
		perror("listInitialize");
		return NULL;
	}

	Node *tail = malloc(sizeof(Node));
	if (tail == NULL) {
		perror("listInitialize");
		free(head);
		return NULL;
	}
	list->head = head;
	list->tail = tail;
	head->next = tail;
	tail->next = tail;
	list->count = 0;
	return list;
}

int listFinalize(List *list) {
	if (list == NULL) {
		fprintf(stderr, "listFinalize:list is NULL\n");
		return -1;
	}

	if (list->head == NULL || list->tail == NULL) {
		fprintf(stderr, "listFinalize: head or tail is NULL.\n");
		return -1;
	}

	while (list->head->next != list->tail) {
		Node *target = list->head->next;
		list->head->next = target->next;
		free(target);
	}
	free(list->head);
	free(list->tail);
	free(list);
	return 0;
}
int listAdd(List *list, void *data) {
	if (list == NULL) {
		fprintf(stderr, "listAdd:list is NULL");
		return -1;
	}

	Node *node = malloc(sizeof(Node));
	if (node == NULL) {
		perror("listAdd");
		return -1;
	}
	node->data = data;
	node->next = list->tail;

	Node *cur = list->head;
	while (cur->next != list->tail)
		cur = cur->next;
	cur->next = node;
	++(list->count);
	return 0;
}

void listDisplay(const List *list, const char *(*displayFunc)(const void *)) {
	if (list == NULL) {
		fprintf(stderr, "listDisplay:list is NULL.\n");
		return;
	}

	system("cls");
	printf("[head]");
	for (Node *node = list->head->next; node != list->tail; node = node->next)
		printf("->[%s]", displayFunc(node->data));
	printf("->[tail]");
	getchar();
}
